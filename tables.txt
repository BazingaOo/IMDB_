CREATE TABLE user(
user_id int not null,
username varchar(50),
password varchar(50),
user_type varchar(50),
gender varchar(50),
birthday date,
PRIMARY KEY(user_id)
);

CREATE TABLE movie(
movie_id int not null,
movie_name varchar(50),
year int,
rate float,
desciption varchar(50),
PRIMARY KEY(movie_id)
);

CREATE TABLE cast_list(
cast_id int not null,
cast_name varchar(50),
cast_desciption varchar(50),
PRIMARY KEY(cast_id)
);


CREATE TABLE genre_list(
genre_id int not null,
genre varchar(50),
PRIMARY KEY(genre_id)
);

CREATE TABLE movie_genre(
movie_id int,
genre_id int,
foreign key(movie_id) references movie(movie_id) on delete cascade on update cascade,
foreign key(genre_id) references genre_list(genre_id) on delete cascade on update cascade
);



CREATE TABLE watched_list(
user_id int,
movie_id int,
foreign key(user_id) references user(user_id) on delete cascade on update cascade,
foreign key(movie_id) references movie(movie_id) on delete cascade on update cascade
);

CREATE TABLE movie_cast(
cast_id int,
movie_id int,
foreign key(cast_id) references cast_list(cast_id) on delete cascade on update cascade,
foreign key(movie_id) references movie(movie_id) on delete cascade on update cascade
);

CREATE TABLE comment(
comment_id int,
user_id int,
movie_id int,
one_rate float,
comment_description varchar(50),
foreign key(user_id) references user(user_id) on delete cascade on update cascade,
foreign key(movie_id) references movie(movie_id) on delete cascade on update cascade
);
